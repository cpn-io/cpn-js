})
export class ProjectTreeOptionsNodeComponent implements OnInit {

  public nodeToArray = nodeToArray;
  public console = console;

  @Input() public optionList: any;
  @Input() public expanded: any;
  @Input() public selected: any;

  ngOnInit() {
  }

}
